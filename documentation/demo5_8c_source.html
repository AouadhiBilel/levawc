196 void heapsort_intro(PQueue pq)
197 {
198  int *pv, retval;
199  Slist tmplst;
200 
201  if (PQUEUEsize(pq) == 0)
202  {
203  prompt_and_pause("Priority queue is empty - nothing to do..");
204  return;
205  }
206 
207  /* Initialize a list - defensive programming... */
208  if ((tmplst = SLISTinit(my_destroy)) == NULL)
209  {
210  printf("\nFatal error - bailing out...\n!");
211  exit(-1);
212  }
213 
214  do
215  {
216  my_clearscrn();
217  printf("--- HEAPSORT INTRODUCTION ---\n");
218 
219  if (PQUEUEpeek(pq))
220  {
221  /* Print the contents of the priority queue... */
222  PQUEUEprint(pq, print);
223  /* Print the list - from the beginning... */
224  printf("=====");
225  printf("\nList: ");
226  SLISTtraverse(tmplst, print, SLIST_FWD);
227  printf ("\n\nNow - move the high priority node, value=%d, from the priority queue...\n", *(int *)PQUEUEpeek(pq));
228  prompt_and_pause("..and insert it - at the front of the list...");
229  /* Extract the top priority node from the queue... */
230  retval = PQUEUEextract(pq, (void **)&pv);
231  /* Defensive programming... */
232  assert(retval == OK);
233  /* Insert this node at the front of a linked list... */
234  retval = SLISTinsnext(tmplst, NULL, pv);
235  /* Defensive programming... */
236  assert(retval == OK);
237  }
238  else
239  {
240  printf("\nHeap: (empty)");
241  printf("\nTree: (empty)\n");
242 
243  /* Print the list - from the beginning... */
244  printf("=====");
245  printf("\nList: ");
246  SLISTtraverse(tmplst, print, SLIST_FWD);
247  prompt_and_pause("\n\nEvidently - the priority queue is now empty..!");
248  break;
249  }
250 
251  } while (1);
252 
253  printf("\n---\nHey - it looks like the priority queue can be used for sorting things.. :-)!");
254  prompt_and_pause("\nTRUE, indeed! Just Google \'Heapsort\' for more..");
255 
256  /* Move data back from list - to priority queue.. */
257  while (SLISTsize(tmplst))
258  {
259  retval = SLISTremnext(tmplst, NULL, (void **)&pv);
260  assert(retval == OK);
261  retval = PQUEUEinsert(pq, pv);
262  assert(retval == OK);
263  }
264 
265  SLISTdestroy(tmplst);
266 }
